                <div>
                  <h3 className="text-sm text-amber-900">
                    <Link href={`/products/${product.id}`}>
                      <span aria-hidden="true" className="absolute inset-0" />
                      {product.name}

                  </h3>
                  <p className="mt-1 text-sm text-amber-700">{product.artisan}</p>
